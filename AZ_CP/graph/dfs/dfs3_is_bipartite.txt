//implementation to check if given graph is bipartite or not
//Bipartite graph : is a graph, where it have no odd-cycle of nodes.

#include<bits/stdc++.h>
using namespace std;

int n, m;
vector<vector<int>> g;
vector<int> vis;
bool is_bipartite = true;

void dfs(int node, int color) {
    vis[node] = color;
    for (auto v: g[node]) {
        if (!vis[v]) {
            dfs(v, 3-color);//doing automatic flipping// can also use "3^color" instead.
        } else if (vis[v] == vis[node]) {//main deciding login, if current node is visited as well as next node is already visited with same color, then not bipartite-graph.
            is_bipartite = false;
        }
    }
}

int main () {
    cin>>n>>m;
    g.resize(n+1);
    
    for (int i=0; i<m; i++) {
        int a, b;
        cin>>a>>b;
        g[a].push_back(b);
        //if graph is undirected, the connect vice-versa also
        g[b].push_back(a);
    }
    
    vis.resize(n+1, 0);
    for (int i=1; i<=n; i++) {
        if (!vis[i]) {
            dfs(i, 1);//pass a color, i.e. 1 and 2
        }
    }
    //printing output
    cout<<is_bipartite<<endl;
    for (int i=1; i<=n; i++) {
        cout<<i<<" : "<<vis[i]<<endl;
    }
    return 0;
}
